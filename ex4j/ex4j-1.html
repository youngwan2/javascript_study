console 에 대한 추가 학습

            - console.group('제1 그룹 시작')
            - console.log('콘솔에서 보이는 내용') : 콘솔 내용
            - console.error('에러입니다!!!') : 에러 표시
            - console.groupEnd('제1 그룹 끝')
            

            - console.group('제2 그룹 시작')
            - console.debug('디버그입니다.!!') : 디버그
            - console.info('정보입니다.') : 정표 표시
            - console.warn('경고입니다.') : 경고 표시
            - console.groupEnd('제2 그룹 끝'): 콘솔 내용 그룹으로 묶어줌
