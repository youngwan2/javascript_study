      const now = dayjs();
      const random = now.get("ms"); // random = 556 

      // 확인과 취소중에 하나를 선택 하세요.
      //조건1) true or flase 로 comfirm 사용하기
      // 확인 : true == 1
      // 취소 : false == 0
      const $confirm = confirm("확인과 취소 중에 하나를 선택 하세요.");
      
      //조건2) % 사용하기
      // 10 % 2 = 0
      // 11 % 2 = 1
      const judgment = random % 2
      
      //조건3) if
      if(judgment ==$confirm){
        alert(`당신이 이겼습니다. 승리를 축하하니다. 랜덤숫자값: ${random}`)
        console.log(judgment,$confirm)
      } else {
        alert(`당신이 졌습니다. 승리를 축하합니다. 랜덤숫자값 : ${random} `)
        console.log(judgment,$confirm)
      }
