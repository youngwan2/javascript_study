자료형 검사

      console.log("     <자료형검사>     ");
      console.log('typeof("") : ', typeof "");
      console.log('typeof("문자열") : ', typeof "문자열");
      console.log('typeof("1234") : ', typeof "1234");
      console.log("typeof(1234) : ", typeof 1234);
      console.log("typeof(true) : ", typeof true);
      console.log('typeof("true") : ', typeof "true");
      console.log("typeof(false) : ", typeof false);
      console.log('typeof("false") : ', typeof "false");
      console.log("typeof(null) : ", typeof null);
      console.log("typeof(undefined) : ", typeof undefined);

    
--출력결과--

      <자료형검사>     
      - typeof("") :  string
      - typeof("문자열") :  string
      - typeof("1234") :  string
      - typeof(1234) :  number
      - typeof(true) :  boolean
      - typeof("true") :  string
      - typeof(false) :  boolean
      - typeof("false") :  string
      - typeof(null) :  object
      - typeof(undefined) :  undefined
