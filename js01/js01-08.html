템플릿 문자열

      const result1 = 3 + 6;
      console.log(`3 + 6 = ${result1}`);
      console.log(`13 x 26 = ${12 * 26}`);
      console.log(`1 =="1" => ${1 == "1"} `);
      console.log(`false == 0 => ${false == 0} `);
      console.log(`false === 0 => ${false === 0} `);
      console.log(`true == 1 => ${true == 1} `);
      console.log(`"" == [] => ${"" == []}`);

      
--- 출력결과 ---

      - 3 + 6 = 9
      - 13 x 26 = 312
      - 1 =="1" => true 
      - false == 0 => true 
      - false === 0 => false 
      - true == 1 => true 
      - "" == [] => true
