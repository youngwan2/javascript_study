복합 대입 연산자

      a = a + 1 === a += 1 
      - const 는 변하지 않는 값을 대입한다. 
      - let 은 변할 가능성이 있는 값을 대입한다.
      

      console.log(`a=${a}, b=${b},c=${c},d=${d},e=${e} 일 때`);
      console.log(`a+=1 => ${(a += 1)}`);
      console.log(`a+=1 => ${(a += 1)}`);
      console.log(`b*=2 => ${(b *= 2)}`);
      console.log(`b*=2 => ${(b *= 2)}`);
      console.log(`b-=2 => ${(b -= 2)}`);
      console.log(`b-=2 => ${(b -= 2)}`);
      console.log(`c/=2 => ${(c /= 2)}`);
      console.log(`c/=2 => ${(c /= 2)}`);
      console.log(`d%=2 => ${(d %= 2)}`);
      console.log(`d-=2 => ${(d -= 2)}`);
      console.log(`e-=2 => ${(d -= 2)}`);
      
      --- 출력결과 ---

        a=10, b=7,c=11,d=13,e=9 일 때
        a+=1 => 11
        a+=1 => 12
        b*=2 => 14
        b*=2 => 28
        b-=2 => 26
        b-=2 => 24
        c/=2 => 5.5
        c/=2 => 2.75
        d%=2 => 1
        d-=2 => -1
        e-=2 => -3
